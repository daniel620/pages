    Basic Selectors #

    /* Element selector */
h1 {
    color: blue;
    font-size: 2rem;
}

/* Class selector */
.container {
    max-width: 1200px;
    margin: 0 auto;
}

/* ID selector */
#header {
    background-color: #333;
    color: white;
}
